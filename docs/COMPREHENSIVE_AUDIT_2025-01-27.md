class PositionSizer:
    @staticmethod
    def calculate_kelly(win_rate: float, avg_win: float, avg_loss: float) -> float:
        """Kelly Criterion: f = (bp - q) / b"""
        b = avg_win / abs(avg_loss)
        p = win_rate
        q = 1 - p
        kelly = (b * p - q) / b
        return max(0, min(kelly, 0.25))  # Cap at 25%
    
    @staticmethod
    def calculate_volatility_based(atr: float, account_risk: float) -> float:
        """Position size based on ATR volatility"""
        return account_risk / atr
```

class DataCache:
    def __init__(self, cache_dir: Path = Path("data/ohlcv")):
        self.cache_dir = cache_dir
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        
    def save(self, symbol: str, interval: str, df: pd.DataFrame):
        filepath = self.cache_dir / symbol / f"{interval}.parquet"
        filepath.parent.mkdir(exist_ok=True)
        df.to_parquet(filepath, compression="snappy")
        
    def load(self, symbol: str, interval: str) -> pd.DataFrame | None:
        filepath = self.cache_dir / symbol / f"{interval}.parquet"
        if filepath.exists():
            return pd.read_parquet(filepath)
        return None
```

    final_capitals = [s['final_capital'] for s in simulations]

    ruin_count = sum(1 for c in final_capitals if c < initial * 0.5)
